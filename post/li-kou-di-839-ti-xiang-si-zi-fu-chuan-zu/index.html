 class Solution {
public:
    vector<int> bcj;
    int find(int x){
        if (bcj[x] == x){
            return x;
        } else{
            return bcj[x] = find(bcj[x]);
        }
    }
    static bool check(const string& sa, const string& sb, int m){
        int num = 0;
        for (int i = 0; i < m; ++i) {
            if (sa[i] != sb[i]){
                num++;
            }
            if (num > 2){
                return false;
            }
        }
        return true;
    }
    int numSimilarGroups(vector<string>& strs) {
        int n = strs.size();
        int m = strs[0].size();
        bcj.resize(n);
        for (int i = 0; i < n; ++i) {
            bcj[i] = i;
        }
        for (int i = 0; i < n; ++i) {
            for (int j = i+1; j < n; ++j) {
                int fi = find(i);
                int fj = find(j);
                if(fi == fj){
                    continue;
                }
                if (check(strs[i],strs[j],m)){
                    bcj[fi] = fj;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (bcj[i] == i){
                ans++;
            }
        }
        return ans;
    }
};
