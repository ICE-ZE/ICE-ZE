class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sumA = 0,sumB = 0;
        for (int i : A) {
            sumA += i;
        }
        for (int i : B) {
            sumB += i;
        }
        vector<int> ans;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = 0; j < B.size(); ++j) {
                if ((sumA - A[i] + B[j]) == (sumB - B[j] + A[i])){
                    ans.push_back(A[i]);
                    ans.push_back(B[j]);
                    break;
                }
            }
            if (ans.size() > 0){
                break;
            }
        }
        return ans;
    }
};

class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B){
        int sumA = 0,sumB = 0;
        for (int i : A) {
            sumA += i;
        }
        for (int i : B) {
            sumB += i;
        }
        sort(B.begin(),B.end());
        int difference = (sumA - sumB);
        vector<int> ans;
        for (int & i : A) {
            int left = 0, right = B.size() - 1;
            while (left < right){
                int mid = (right + left) / 2;
                if (B[mid] > (i + difference/2)){
                    right = mid - 1;
                } else if(B[mid] < (i + difference/2)){
                    left = mid + 1;
                } else{
                    ans.push_back(i);
                    ans.push_back(B[mid]);
                    return ans;
                }
            }
        }
        return ans;
    }
};
